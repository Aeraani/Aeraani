1    package com.multi.importer;
2    
3    import com.opencsv.CSVReader;
4    
5    import java.io.InputStreamReader;
6    import java.nio.charset.StandardCharsets;
7    import java.sql.Connection;
8    import java.sql.DriverManager;
9    import java.sql.PreparedStatement;
10   
11   public class CsvImporter {
12       static final String URL = "jdbc:mysql://localhost:3306/travel_db?useSSL=false&serverTimezone=Asia/Seoul&characterEncoding=utf8";
13       static final String USER = "apple";
14       static final String PASS = "apple";
15   
16       public static void main(String[] args) throws Exception {
17           try(Connection connection = DriverManager.getConnection(URL, USER, PASS)) {
18               connection.setAutoCommit(false);
19               String sql = "INSERT INTO travel(district, title, description, address, phone) VALUES(?,?,?,?,?)";
20               try (PreparedStatement ps = connection.prepareStatement(sql);
21                    //travel.csv를 읽어오기 위한 reder코드 설정
22                    CSVReader reader = new CSVReader(new InputStreamReader(CsvImporter.class.getResourceAsStream("/travel.csv"),
23                            StandardCharsets.UTF_8))) {
24                   String[] row;
25                   boolean headerSkipped = false;
26                   while ((row = reader.readNext()) != null) {
27                       if (!headerSkipped) { headerSkipped = true; continue; } // 헤더 스킵
28                       ps.setString(1, row[1]); // district
29                       ps.setString(2, row[2]); // title
30                       ps.setString(3, row[3]); // description
31                       ps.setString(4, row[4]); // address
32                       ps.setString(5, row[5]); // phone
33                       ps.addBatch();
34                   }
35                   ps.executeBatch();
36                   connection.commit();
37                   System.out.println("CSV import 완료!");
38               }
39           }catch (Exception e) {
40               e.printStackTrace();
41           }
42   
43       }
44   }
